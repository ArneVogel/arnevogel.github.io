The code for the summed newton cotes formulas in matlab.

Trapezoid rule

function t = trapezoid_rule(a,b,n)
  t = 0;
  x = linspace(a,b,n);
  h = x(2)-x(1);
  for i = 1:length(x)-1
    t = t + (h/2 * (f(x(i)) + f(x(i+1))));
  end
end

Simpson’s rule

function s = simpson_rule(a,b,n)
  s = 0;
  x = linspace(a,b,n);
  h = x(2)-x(1);
  for i = 1:length(x)-1
    s = s + (f(x(i)) + 4* f((x(i)+x(i+1))/2) + f(x(i+1)));
  end
  s = s * h/6;
end

3⁄8 rule

function r = regel38(a,b,n)
  r = 0;
  x = linspace(a,b,n);
  h = x(2)-x(1);
  for i = 1:length(x)-1
    r = r + ( f(x(i)) + 3* f(((x(i) + x(i+1))*(4/3)/3)) + 3* f(((x(i) + x(i+1))*(5/3))/3) + f(x(i+1)));
  end
  r = r * h/8;
end
